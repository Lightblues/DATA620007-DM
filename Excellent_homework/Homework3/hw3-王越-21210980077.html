### 第五题
dummy_district = pd.get_dummies(df['district']).drop(columns = '昌平区') ###以昌平区为基准把区域one-hot编码
dummy_shop_type = pd.get_dummies(df['shop_type']).drop(columns = '果蔬生鲜') ###以果蔬生鲜为基准把店铺类型one-hot编码
tmp_df = df.copy()
tmp_df.loc[:,['delivery_mode']] = tmp_df['delivery_mode'] == '商家配送' 
###把表格中配送模式为商家配送的转化为True,蜂鸟专送为false

columns = ['log_average_cost','delivery_mode','log_float_delivery_fee','log_float_minimum_order_amount',\
           'order_lead_time','保','票','manjian_discount','other_activities','new_user','open_time','open08',\
          'open12','open18','open23','is_new','is_premium'] ###所有可取特征的列表
data = pd.DataFrame(tmp_df.loc[:,columns],dtype = 'float64') ###把表中所有布尔值转为1和0浮点数
y_train = df['log_recent_order_num'] ###训练集的标签
select = [] ###选择的特征列表
bic_min = float('+inf') ###当前bic值
count = 0 ###计数器
while columns:
    bics = [] ###初始化bic列表
    for i in columns:
        tmp_vari = select +[i] ###每次选择的特征，包含之前已经选择过的和columns中剩下的一个特征
        X_train = data.loc[:,tmp_vari].join(dummy_district).join(dummy_shop_type) ###把这轮选的特征和哑变量拼接
        X_train = sm.add_constant(X_train) ###加入常数项
        model = sm.OLS(y_train,X_train).fit() ###训练模型
        bics.append((model.bic,i)) ###往bic列表添加(bic值,本次所选特征)的元组
    bics.sort() ###按照bic值从小到大排序
    mini = bics.pop(0) ###选取bic值最小的一个元组
    if mini[0] < bic_min: ###如果该bic值小于当前bic值
        select.append(mini[1]) ###往选择的特征列表里加上该特征
        columns.remove(mini[1]) ###可选特征列表移除该特征，下次循环的时候就不会重复选取特征了
        bic_min = mini[0] ###更新当前bic值
        print(F"Round: {count}, Bic: {round(bic_min,3)}, Feature: {mini[1]}") ###打印这轮结果
        count += 1 ###计数器++
    else:
        print('finished') ###如果所有bic值都大于当前bic值，说明剩下的特征都不该被选择，逐步选择特征结束。
        break

另一方面，新店对月销量有着负面的影响。一家新开的店相比老店的对数月销量会少1.47，这是因为平台推荐系统的冷启动问题，新店的曝光率会比较低，所以销售量就会少。配送费对月销量的负面影响也较大，平均对数配送费每上升1，对数月销量就会下降0.31。这是因为配送费增加了消费者的成本，并且配送费越贵说明距离越远，配送时间会越久，消费者一般会倾向于点较近、配送费便宜（甚至免配送费）的外卖。¶
